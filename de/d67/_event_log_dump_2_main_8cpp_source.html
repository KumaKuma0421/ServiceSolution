302 LPTSTR GetMessageString(HANDLE hResource, DWORD dwMessageID, DWORD dwArgc, LPTSTR lptszArgv)
303 {

306  FORMAT_MESSAGE_FROM_SYSTEM |

311 
312  // The insertion strings appended to the end of the event record
313  // are an array of strings; however, FormatMessage requires
314  // an array of addresses. Create an array of DWORD_PTRs based on
315  // the count of strings. Assign the address of each string
316  // to an element in the array (maintaining the same order).

327  lptszString += _tcslen(lptszString) + 1;

329  }

333  _tprintf(_T("Failed to allocate memory for the insert string array.\n"));
334  }
335  }
336 
337  if (!::FormatMessage(dwFormatFlags,
338  hResource,
339  dwMessageID,
340  0,
341  (LPTSTR)&lptszMessage,
342  0,
343  (va_list*)lpdwArgs))

345  _tprintf(_T("Format message failed with %lu\n"), ::GetLastError());
346  }

354 // If the message string contains parameter insertion strings (for example, %%4096),
355 // you must perform the parameter substitution yourself. To get the parameter message
356 // string, call FormatMessage with the message identifier found in the parameter insertion
357 // string (for example, 4096 is the message identifier if the parameter insertion string
358 // is %%4096). You then substitute the parameter insertion string in the message
359 // string with the actual parameter message string.
360 DWORD ApplyParameterStringsToMessage(LPCTSTR lptszMessage, LPTSTR& lptszFinalMessage)
361 {
362  DWORD dwStatus = ERROR_SUCCESS;
363  DWORD dwParameterCount = 0; // Number of insertion strings found in lptszMessage
364  size_t cbBuffer = 0; // Size of the buffer in bytes
365  size_t cchBuffer = 0; // Size of the buffer in characters
366  size_t cchParameters = 0; // Number of characters in all the parameter strings
367  size_t cch = 0;
368  DWORD i = 0;
369  LPTSTR* lptszStartingAddresses = nullptr; // Array of pointers to the beginning of each parameter string in lptszMessage
370  LPTSTR* lptszEndingAddresses = nullptr; // Array of pointers to the end of each parameter string in lptszMessage
371  LPDWORD lpdwParameterIDs = nullptr; // Array of parameter identifiers found in lptszMessage
372  LPTSTR* lptszParameters = nullptr; // Array of the actual parameter strings
373  LPTSTR lptszTempMessage = (LPTSTR)lptszMessage;
374  LPTSTR lptszTempFinalMessage = nullptr;
375 
376  // Determine the number of parameter insertion strings in lptszMessage.
377  while (lptszTempMessage = ::_tcschr(lptszTempMessage, L'%'))
378  {
379  dwParameterCount++;
380  lptszTempMessage++;
381  }
382 
383  // If there are no parameter insertion strings in lptszMessage, return.
384  if (0 == dwParameterCount)
385  {
386  lptszFinalMessage = nullptr;
387  goto cleanup;
388  }
389 
390  // Allocate an array of pointers that will contain the beginning address
391  // of each parameter insertion string.
392  cbBuffer = sizeof(LPTSTR) * dwParameterCount;
393  lptszStartingAddresses = (LPTSTR*)::malloc(cbBuffer);
394  if (nullptr == lptszStartingAddresses)
395  {
396  _tprintf(_T("Failed to allocate memory for lptszStartingAddresses.\n"));
397  dwStatus = ERROR_OUTOFMEMORY;
398  goto cleanup;
399  }
400 
401  ::RtlZeroMemory(lptszStartingAddresses, cbBuffer);
402 
403  // Allocate an array of pointers that will contain the ending address (one
404  // character past the of the identifier) of the each parameter insertion string.
405  lptszEndingAddresses = (LPTSTR*)::malloc(cbBuffer);
406  if (nullptr == lptszEndingAddresses)
407  {
408  _tprintf(_T("Failed to allocate memory for lptszEndingAddresses.\n"));
409  dwStatus = ERROR_OUTOFMEMORY;
410  goto cleanup;
411  }
412 
413  ::RtlZeroMemory(lptszEndingAddresses, cbBuffer);
414 
415  // Allocate an array of pointers that will contain pointers to the actual
416  // parameter strings.
417  lptszParameters = (LPTSTR*)::malloc(cbBuffer);
418  if (nullptr == lptszParameters)
419  {
420  _tprintf(_T("Failed to allocate memory for lptszEndingAddresses.\n"));
421  dwStatus = ERROR_OUTOFMEMORY;
422  goto cleanup;
423  }
424 
425  ::RtlZeroMemory(lptszParameters, cbBuffer);
426 
427  // Allocate an array of DWORDs that will contain the message identifier
428  // for each parameter.
429  lpdwParameterIDs = (DWORD*)::malloc(cbBuffer);
430  if (nullptr == lpdwParameterIDs)
431  {
432  _tprintf(_T("Failed to allocate memory for lpdwParameterIDs.\n"));
433  dwStatus = ERROR_OUTOFMEMORY;
434  goto cleanup;
435  }
436 
437  ::RtlZeroMemory(lpdwParameterIDs, cbBuffer);
438 
439  // Find each parameter in lptszMessage and get the pointer to the
440  // beginning of the insertion string, the end of the insertion string,
441  // and the message identifier of the parameter.
442  lptszTempMessage = (LPTSTR)lptszMessage;
443  while (lptszTempMessage = ::_tcschr(lptszTempMessage, L'%'))
444  {
445  if (::isdigit(*(lptszTempMessage + 1)))
446  {
447  lptszStartingAddresses[i] = lptszTempMessage;
448 
449  lptszTempMessage++;
450  lpdwParameterIDs[i] = (DWORD)::_wtoi(lptszTempMessage);
451 
452  while (::isdigit(*++lptszTempMessage))
453  ;
454 
455  lptszEndingAddresses[i] = lptszTempMessage;
456 
457  i++;
458  }
459  }
460 
461  // For each parameter, use the message identifier to get the
462  // actual parameter string.
463  for (DWORD i = 0; i < dwParameterCount; i++)
464  {
465  lptszParameters[i] = GetMessageString(g_hMessageResource, lpdwParameterIDs[i], 0, nullptr); // TODO:g_hMessageResourceは暫定措置
466  if (nullptr == lptszParameters[i])
467  {
468  _tprintf(_T("GetMessageString could not find parameter string for insert %lu.\n"), i);
469  dwStatus = ERROR_INVALID_PARAMETER;
470  goto cleanup;
471  }
472 
473  cchParameters += ::_tcslen(lptszParameters[i]);
474  }
475 
476  // Allocate enough memory for lptszFinalMessage based on the length of lptszMessage
477  // and the length of each parameter string. The lptszFinalMessage buffer will contain
478  // the completed parameter substitution.
479  lptszTempMessage = (LPTSTR)lptszMessage;
480  cbBuffer = (_tcslen(lptszMessage) + cchParameters + 1) * sizeof(WCHAR);
481  lptszFinalMessage = (LPTSTR)::malloc(cbBuffer);
482  if (nullptr == lptszFinalMessage)
483  {
484  _tprintf(_T("Failed to allocate memory for lptszFinalMessage.\n"));
485  dwStatus = ERROR_OUTOFMEMORY;
486  goto cleanup;
487  }
488 
489  ::RtlZeroMemory(lptszFinalMessage, cbBuffer);
490  cchBuffer = cbBuffer / sizeof(WCHAR);
491  lptszTempFinalMessage = lptszFinalMessage;
492 
493  // Build the final message string.
494  for (DWORD i = 0; i < dwParameterCount; i++)
495  {
496  // Append the segment from lptszMessage. In the first iteration, this is "8 " and in the
497  // second iteration, this is " = 2 ".
498  _tcsncpy_s(lptszTempFinalMessage, cchBuffer, lptszTempMessage, cch = (lptszStartingAddresses[i] - lptszTempMessage));
499  lptszTempMessage = lptszEndingAddresses[i];
500  cchBuffer -= cch;
501 
502  // Append the parameter string. In the first iteration, this is "quarts" and in the
503  // second iteration, this is "gallons"
504  lptszTempFinalMessage += cch;
505  _tcscpy_s(lptszTempFinalMessage, cchBuffer, lptszParameters[i]);
506  cchBuffer -= cch = _tcslen(lptszParameters[i]);
507 
508  lptszTempFinalMessage += cch;
509  }
510 
511  // Append the last segment from lptszMessage, which is ".".
512  _tcscpy_s(lptszTempFinalMessage, cchBuffer, lptszTempMessage);
513 
514 cleanup:
515 
516  if (ERROR_SUCCESS != dwStatus)
517  lptszFinalMessage = (LPTSTR)lptszMessage;
518 
519  if (lptszStartingAddresses)
520  ::free(lptszStartingAddresses);
521 
522  if (lptszEndingAddresses)
523  ::free(lptszEndingAddresses);
524 
525  if (lpdwParameterIDs)
526  ::free(lpdwParameterIDs);
527 
528  for (DWORD i = 0; i < dwParameterCount; i++)
529  {
530  if (lptszParameters[i])
531  ::LocalFree(lptszParameters[i]);
532  }
533 
534  return dwStatus;
535 }
